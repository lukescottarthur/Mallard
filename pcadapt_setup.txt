#This code is used to merge vcf.gz files, then convert to bed file for input to pcadapt R package.

conda activate test_env

#view vcf file header
bcftools view -h *.vcf.gz

##change headers
#extract header
bcftools view -h chr17.GFM.withId.vcf.gz > headerGFM.txt   
bcftools view -h chr17.WMA.withID.vcf.gz > headerWMA.txt 
bcftools view -h KC_chr17.withID.vcf.gz > headerKC.txt 

#move txt file
mv headerGFM.txt /home/las80898/Mallard/
mv headerWMA.txt /home/las80898/Mallard/
mv headerKC.txt /home/las80898/Mallard/

#manually edit header in vscode or choice of text editor
##FORMAT=<ID=MQ,Number=1,Type=Float,Description="Average mapping quality">  
note: i didn't change to 'FORMAT', just changed 'Type=Float'

#move txt file back
mv /home/las80898/Mallard/headerGFM.txt /home/las80898/mallard_data 
mv /home/las80898/Mallard/headerWMA.txt /home/las80898/mallard_data 
mv /home/las80898/Mallard/headerKC.txt /home/las80898/mallard_data 

#reheader the file
bcftools reheader -h headerGFM.txt -o chr17.GFMIDreheader.vcf.gz chr17.GFM.withId.vcf.gz   
bcftools reheader -h headerWMA.txt -o chr17.WMAIDreheader.vcf.gz chr17.WMA.withID.vcf.gz   
bcftools reheader -h headerKC.txt -o chr17.KCIDreheader.vcf.gz KC_chr17.withID.vcf.gz
##

# Filter KC (or whatever for multiallelic variants)
bcftools view --max-alleles 2 chr17.KCIDreheader.vcf.gz -Oz -o chr17.KCIDreheaderfiltered.vcf.gz 

#create index files
bcftools index chr17.GFMIDreheader.vcf.gz 
bcftools index chr17.WMAIDreheader.vcf.gz 
bcftools index chr17.KCIDreheaderfiltered.vcf.gz



##merge vcf files

# wild & game farm
bcftools merge chr17.GFMIDreheader.vcf.gz chr17.WMAIDreheader.vcf.gz -Oz -o chr17merge.vcf.gz

#convert vcf to bed
plink2 --vcf chr17merge.vcf.gz --make-bed --out chr17

# wild and khaki campbell
bcftools merge chr17.WMAIDreheader.vcf.gz chr17.KCIDreheaderfiltered.vcf.gz -Oz -o chr17merge.WMAKC.vcf.gz  
# convert to bed
plink2 --vcf chr17merge.WMAKC.vcf.gz --make-bed --out chr17WMAKC

# game farm and khaki campbell
bcftools merge chr17.GFMIDreheader.vcf.gz chr17.KCIDreheaderfiltered.vcf.gz -Oz -o chr17merge.GFMKC.vcf.gz  
# convert to bed
plink2 --vcf chr17merge.GFMKC.vcf.gz --make-bed --out chr17GFMKC
