This is a workflow to get started with diploS/HIC
see: https://github.com/kr-colab/diploSHIC/wiki/A-soup-to-nuts-example

# Before starting, make sure to change file paths in .sh script to match your file path (three spots in header, one in code)

# create environment (this will take a few minutes)
conda env create -f diplo_env.yaml

# download (clone) diploshic repository from github
git clone https://github.com/kr-colab/diploSHIC.git

# navigate into diploshic directory 
cd diploSHIC

# download example dataset
wget http://sesame.uoregon.edu/~adkern/diploSHIC/exampleApplication.tar.gz && tar zxvf exampleApplication.tar.gz

# run sbatch script (big job, 16GB, 8 CPUS, runs for about 2 hours on cluster)
sbatch diploshic_test_1.sh

# view output
less mossie.preds
